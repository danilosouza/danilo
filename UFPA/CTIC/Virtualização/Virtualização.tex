\section{Virtualização}

	\subsection{Simulação, Emulação e Virtualização}
		\begin{frame}{Definições}
			\begin{itemize}
				\item Simulação
					\begin{itemize}
						\item `` \textit{É o ato de representar com semelhanças certos aspectos de; imitar 							ou fingir.} '' [Houassis]
						\item Simulação computacional
					\end{itemize}
				\pause
				\item Emulação
					\begin{itemize}
						\item `` \textit{É o esforço para imitar ou tentar seguir o exemplo de alguém} '' 								[Houassis]
						\item Um programa imita outro programa ou dispositivo
					\end{itemize}
				\pause				
				\item Virtualização
					\begin{itemize}
						\item  `` \textit{Virtualização:
							É um framework ou metodologia para dividir os recursos de um computador em 
							múltiplos ambientes de execução, aplicando um, ou mais conceitos, e 										tecnologias como particionamento de software ou hardware, tempo compartilhado, 							simulação completa ou parcial da máquina, emulação e qualidade de serviços.} 								'' [Singh, 2008]
					\end{itemize}
			\end{itemize}
		\end{frame}

	\subsection{Virtualização Total X Paravirtualização}
			\begin{frame}{Virtualização Total}
				\begin{itemize}
					\item Consiste em prover uma réplica do hardware original para a máquina hóspede, de 							tal forma que esta não saiba que está sendo virtualizada.
					\pause
					\item Vantagens
						\begin{itemize}
							\item Não precisa alterar o SO hóspede
							\item Suporte de hardware
						\end{itemize}
					\item Desvantangens
					\pause
						\begin{itemize}
							\item Difícil suporte a grande variedade de dispositivos
							\item Subutilização de recursos
							\item Instruções devem ser testadas pela VMM (queda de desempenho)
							\item Converter o espaço de endereçamento do hóspede (queda de desempenho)
						\end{itemize}
				\end{itemize}								
			\end{frame}
			
			\begin{frame}{Paravirtualização}
				\begin{itemize}
					\item Surge como alternativa à Virtualização Total
					\item O SO chama a VMM sempre que for executar uma instrução sensível
					\item O SO hóspede possui drivers próprios
					\pause					
					\item Vantagens
						\begin{itemize}
							\item Melhor desempenho
							\item Utilização máxima dos recursos de hardware
						\end{itemize}
					\item Desvantagens
						\begin{itemize}
							\item Necessita alteração de SO tanto no hospedeiro quanto no hóspede
						\end{itemize}
				\end{itemize}
			\end{frame}
