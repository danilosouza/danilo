\section{Introdução à Firewall}

	\begin{frame}
		\begin{block}{Definição}
			Um conjunto de aplicações que realiza controle de entrada e saída de uma determinada rede, sobre qualquer uma das camadas da pilha 					TCP/IP
		\end{block}
		\begin{itemize}
			\item \textit{NetFilter} - Pacotes
				\begin{itemize}
					\item \textit{Stateless}
						\begin{itemize}
							\item Toma decisões baseado somente no pacote atual
							\item Muito utilizado em roteadores simples
							\item Baixo custo computacional
						\end{itemize}
					\item \textit{Statefull}
						\begin{itemize}
							\item Toma decisões baseado em informaões armazenadas
							\item Mais utilizado em servidores
							\item Alto custo computacional
						\end{itemize}			
				\end{itemize}
			\item Squid (Proxy)
				\begin{itemize}
					\item Filtro de aplicação
					\item Bloqueio de url\textquotesingle s e palavras chave
				\end{itemize} 
		\end{itemize}
		
		\begin{alertblock}{Importante}
			Um firewall mal configurado é pior do que não ter um firewall	
		\end{alertblock}
	\end{frame}
	
	\begin{frame}{Proteção}
		\begin{itemize}
			\item Firewall de \textit{host}
				\begin{itemize}
					\item Protege somente a máquina local
					\item Regras mais robustas
				\end{itemize}
			\item Firewall de rede
				\begin{itemize}
					\item Protege a(s) rede(s) à qual pertence
					\item Regras mais simples
					\item \alert {Pode se tornar um gargalo}
				\end{itemize}
		\end{itemize}
	\end{frame}
