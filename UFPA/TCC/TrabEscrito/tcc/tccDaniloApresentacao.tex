\begin{frame}{Segmentação de Imagens - Definição}

	\begin{variableblock}{Definição (Gonzales et. al)}{bg=blue!15,fg=black}{bg=blue!70,fg=white}
  		Consiste em dividir uma imagem em sub-imagens formadas pelos objetos/regiões da imagem original. A divisão não deve ser realizada além do necessário para detectar os objetos e/ou regiões relevantes para a aplicação.
	\end{variableblock}	

	\begin{enumerate}
		\item $ \bigcup_{i=1}^{n}R_{i} = R $.
		\item $ R_{i}$ é um conjunto conectado, $i = 1, 2,\dots, n $.
		\item $ R_{i} \cap R_{j} = \emptyset $ para todo $i$ e $j$, onde $ i \neq j $.
		\item $ Q(R_{i}) = VERDADEIRO\ para\  i = 1, 2,\dots, n $.
		\item $ Q(R_{i} \cup R_{j}) = FALSO $, onde $ i \neq j$, onde $Q(R_{k})$ representa a função e/ou propriedade (cor, textura, geometria) que deve ser satisfeita pelos \textit{pixels} pertencentes à $R_{k}$.
	\end{enumerate}

\end{frame}


\begin{frame}{Segmentação de Imagens - Técnicas}
			\begin{itemize}
				\item Limiar (\textit{threshold})
					\begin{itemize}
						\item Forma mais simples de segmentação
						\begin{equation}
								g(x,y) = 		
								\begin{cases}
								 	0, \text{se } p(x,y) < T \\
									1, \text{se } p(x,y) > T
								\end{cases}
						\end{equation}
					\end{itemize}
				\item Dividir e mesclar
					\begin{enumerate}
						\item Dividir a imagem em regiões de tamanhos iguais
						\item Verifica se as novas regiões obedecem o critério estabelecido
						\item Mescla as regiões em comum
						\item Repete os passo (1) e (2) até que não haja novas imagens
					\end{enumerate}
				\item Detecção de bordas
					\begin{itemize}
						\item Calcula o gradiente de intensidade
						\item Bordas possuem mudanças abruptas de intensidade
					\end{itemize}
		\end{itemize}
	
\end{frame}

\section{A técnica estudada}

\begin{frame}{A técnica estudada (Protiere e Sapiro)}

	\begin{variableblock}{Princípio}{bg=blue!15,fg=black}{bg=blue!70,fg=white}
  		Classificar um \textit{pixel} com base na distância e na intensidade, utilizando marcações do usuário (heurística).
	\end{variableblock}	

	\begin{variableblock}{Robustez}{bg=green!15,fg=black}{bg=green!90,fg=white}
		O algoritmo suporta a marcação de regiões não-uniformes, sendo capaz de realizar segmentações precisas em imagens complexas.
	\end{variableblock}	

	\begin{variableblock}{O Problema}{bg=red!15,fg=black}{bg=red!80,fg=white}
		O tempo de execução do algoritmo é a sua principal desvantagem, sendo portanto o objeto de estudo deste trabalho.
	\end{variableblock}		

\end{frame}

	\begin{figure}[!h]
		\centering

\begin{frame}{O algoritmo}
\begin{multicols}{2}
	\begin{enumerate}
		\item Marcar as regiões de interesse.
		\item Guardar valores e posição destes pontos e calcular as FDP's de cada região.
		\item Calcular os canais.
		\item Repetir o passo 2 para cada canal.
		\item Calcular o peso de cada canal (adaptativo).
		\item Calcular o peso da distância Geodésica.
		\item Se $T_{re-amostragem} < 1$, então 8. Se $T_{re-amostragem} = 1$, então 9.  
		\item Re-amostrar os \textit{pixels} das regiões de interesse.
		\item Calcular a menor distância de cada \textit{pixel} para todas as sub-regiões.
		\item Calcular a probabilidade de cada \textit{pixel} para todas as sub-regiões.
	\end{enumerate}
\end{multicols}
\end{frame}



\section{Metodologia utilizada}


\begin{frame}{Método de avaliação}

	Foram escolhidas duas métricas principais para avaliação:

	\begin{itemize}
		\item Tempo
			\begin{itemize}
				\item Tempo relativo (Equação \ref{eq:tempo_relativo})
				\item Funções \textit{tic/toc} e \textit{etime} do $MatLab^{\copyright}$
			\end{itemize}
		\item Erro de classificação relativo
			\begin{itemize}
				\item \% de \textit{pixels} classificados incorretamente (Equação \ref{eq:erro_relativo}).
			\end{itemize}
	\end{itemize}
	
		\begin{equation}
		T_{r}^{i} = 100 - \bigg( 100\frac{T_{\text{total}}^{i}}{T_{\text{total}}^{\textit{Full Set}}}\ \bigg) ,\ onde\ i = 1, 10, 50. 
		\label{eq:tempo_relativo}
		\end{equation}
		
		\begin{equation}
			\text{Erro} = \frac{N_{\textit{pixels}\ \text{errados}}}{N_{\text{total}\ \text{de}\ \textit{pixels}}},\ \quad
			N_{\textit{pixels}\ \text{errados}} = I_{100\%} - I_{i}\ \forall\ i =  50\%, 10\%\ \text{e}\ 1\% 
		\label{eq:erro_relativo}
		\end{equation}


\end{frame}	

\begin{frame}{Modificação realizada}

	\begin{variableblock}{Redução do espaço de busca}{bg=blue!15,fg=black}{bg=blue!70,fg=white}
		Reduzir o número de cálculos para encontrar a menor distância sem perder informação.
		\linebreak
		  		
  		Re-amostragem uniforme dos \textit{pixels} das regiões de interesse usando taxas de re-amostragem de $ 50\%, 10\%, e\ 1\%$.
  		
	\end{variableblock}	


\end{frame}
